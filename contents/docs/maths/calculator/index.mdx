## Overview

The calculator function performs basic arithmetic operations including addition, subtraction, multiplication, and division. It provides a simple interface for mathematical computations.

## Usage

```python
from taizun import calculator
num1, num2 = 10, 5
operation_type = 'add'
result = calculator(num1, num2, operation_type)
print(result)
```

## Theory

Basic arithmetic operations form the foundation of mathematical computation:

1. **Addition (+)**: Combines two numbers to produce a sum. Commutative and associative operation.

2. **Subtraction (-)**: Finds the difference between two numbers. Neither commutative nor associative.

3. **Multiplication (ร)**: Repeated addition of one number by another. Commutative and associative operation.

4. **Division (รท)**: Determines how many times one number is contained within another. Neither commutative nor associative.

In computer implementations, special considerations include:
- Handling division by zero errors
- Managing floating-point precision
- Dealing with integer overflow
- Supporting different number types (integers, floats, complex numbers)

Modern calculators often implement additional features like:
- Order of operations (PEMDAS/BODMAS)
- Parenthetical expressions
- Scientific functions (exponents, roots, logarithms)
- Memory functions for storing intermediate results

## Real-World Example

Financial applications and budgeting tools use calculator functions to perform complex financial calculations, helping individuals and businesses make informed monetary decisions:

```python
from taizun import calculator

# Financial calculator for loan analysis
def loan_calculator(principal, annual_rate, years):
    """
    Calculate monthly payment and total interest for a loan
    """
    monthly_rate = annual_rate / 12 / 100
    months = years * 12
    
    # Monthly payment formula: P * (r(1+r)^n) / ((1+r)^n - 1)
    numerator = calculator(principal, calculator(monthly_rate, calculator(1 + monthly_rate, months, 'power'), 'multiply'), 'multiply')
    denominator = calculator(calculator(1 + monthly_rate, months, 'power'), 1, 'subtract')
    
    monthly_payment = calculator(numerator, denominator, 'divide')
    total_payment = calculator(monthly_payment, months, 'multiply')
    total_interest = calculator(total_payment, principal, 'subtract')
    
    return {
        "monthly_payment": round(monthly_payment, 2),
        "total_payment": round(total_payment, 2),
        "total_interest": round(total_interest, 2),
        "interest_percentage": round(calculator(total_interest, principal, 'divide') * 100, 2)
    }

# Example: $300,000 mortgage at 4.5% for 30 years
mortgage = loan_calculator(300000, 4.5, 30)

print("Mortgage Analysis:")
print("=" * 20)
print(f"Loan Amount: $300,000")
print(f"Interest Rate: 4.5%")
print(f"Loan Term: 30 years")
print(f"Monthly Payment: ${mortgage['monthly_payment']:,}")
print(f"Total Payment: ${mortgage['total_payment']:,}")
print(f"Total Interest: ${mortgage['total_interest']:,}")
print(f"Interest Percentage: {mortgage['interest_percentage']}%")

# Investment return calculator
def investment_return_calculator(principal, annual_rate, years, contributions=0):
    """
    Calculate future value of an investment with regular contributions
    Future Value = P(1+r)^t + C[((1+r)^t - 1) / r]
    """
    # Future value of initial principal
    fv_principal = calculator(principal, calculator(1 + annual_rate/100, years, 'power'), 'multiply')
    
    # Future value of regular contributions
    if contributions > 0 and annual_rate > 0:
        rate_decimal = annual_rate / 100
        compound_factor = calculator(1 + rate_decimal, years, 'power')
        fv_contributions = calculator(contributions, calculator(calculator(compound_factor, 1, 'subtract'), rate_decimal, 'divide'), 'multiply')
    else:
        fv_contributions = 0
    
    total_fv = calculator(fv_principal, fv_contributions, 'add')
    
    return {
        "future_value": round(total_fv, 2),
        "principal_return": round(fv_principal, 2),
        "contribution_return": round(fv_contributions, 2),
        "total_contributions": contributions * years,
        "investment_gains": round(calculator(total_fv, calculator(principal, contributions * years, 'add'), 'subtract'), 2)
    }

# Example: $10,000 initial investment, 7% annual return, 25 years, $200 monthly contributions
investment = investment_return_calculator(10000, 7, 25, 200 * 12)

print("\nInvestment Analysis:")
print("=" * 20)
print(f"Initial Investment: $10,000")
print(f"Annual Return: 7%")
print(f"Investment Period: 25 years")
print(f"Annual Contributions: ${200 * 12:,}")
print(f"Future Value: ${investment['future_value']:,}")
print(f"Investment Gains: ${investment['investment_gains']:,}")
print(f"Total Contributions: ${investment['total_contributions']:,}")

# Business expense analysis
def business_expense_analyzer(monthly_expenses, growth_rate, months):
    """
    Calculate projected business expenses with growth
    """
    total_expenses = 0
    current_monthly = monthly_expenses
    
    for month in range(1, months + 1):
        total_expenses = calculator(total_expenses, current_monthly, 'add')
        # Apply monthly growth rate
        current_monthly = calculator(current_monthly, calculator(1 + growth_rate/100, 1, 'add'), 'multiply')
    
    return round(total_expenses, 2)

# Example: $5,000 monthly expenses with 2% monthly growth over 12 months
expenses = business_expense_analyzer(5000, 2, 12)

print("\nBusiness Expense Projection:")
print("=" * 28)
print(f"Initial Monthly Expenses: $5,000")
print(f"Monthly Growth Rate: 2%")
print(f"Projection Period: 12 months")
print(f"Total Projected Expenses: ${expenses:,}")
```
